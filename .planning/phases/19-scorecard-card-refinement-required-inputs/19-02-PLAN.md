**multi-select** (`single_select` falsy or absent):
```js
structured_data[fieldKey] = [
  { value: 'competitor',  answer: '...', next_steps: '...' },
  { value: 'certification', answer: '...', next_steps: '...' }
];
```

**single-select** (`single_select: true`):
```js
structured_data[fieldKey] = 'documented';   // the picked value
structured_data[`${fieldKey}__documented`] = { description: 'rewrote intake checklist' };  // per_selection_fields under sibling key
```

Why the dual-key shape for single_select: keeps per_selection_fields out of the same key as the chosen value (so a renderer doesn't need to special-case "is this a string or an object?"), and `__` prefix avoids any collision with real schema field keys. The Wave 0 validateStructuredFields multi_choice branch is already coded against this contract — see Scorecard.jsx after Wave 0 lands.

```jsx
if (field.type === 'multi_choice') {
  const options = Array.isArray(field.options) ? field.options : [];
  const perFields = Array.isArray(field.per_selection_fields) ? field.per_selection_fields : [];
  const isSingle = !!field.single_select;
  const currentValue = data?.[field.key];

  // Build a normalized list of "active selections" for rendering per_selection sub-blocks.
  const activeSelections = isSingle
    ? (typeof currentValue === 'string' && currentValue.length > 0 ? [currentValue] : [])
    : (Array.isArray(currentValue) ? currentValue.map((sel) => sel.value) : []);

  const toggleSelection = (optionValue) => {
    if (isSingle) {
      // Single-select: write value as string; clear sibling per-selection bucket for any
      // previously selected option to avoid stale data persisting across changes.
      const next = { ...data };
      // Clear any previous sibling buckets `${field.key}__<oldValue>` not equal to optionValue.
      for (const opt of options) {
        const siblingKey = `${field.key}__${opt.value}`;
        if (opt.value !== optionValue && siblingKey in next) delete next[siblingKey];
      }
      next[field.key] = optionValue;
      onDataChange(next);
      return;
    }
    // Multi-select: array of {value, ...per_field_values}
    const arr = Array.isArray(currentValue) ? currentValue.slice() : [];
    const idx = arr.findIndex((sel) => sel.value === optionValue);
    if (idx === -1) {
      const fresh = { value: optionValue };
      for (const pf of perFields) fresh[pf.key] = '';
      arr.push(fresh);
    } else {
      arr.splice(idx, 1);
    }
    onDataChange({ ...data, [field.key]: arr });
  };

  const updatePerField = (optionValue, perKey, newVal) => {
    if (isSingle) {
      const siblingKey = `${field.key}__${optionValue}`;
      const prev = data?.[siblingKey] && typeof data[siblingKey] === 'object' ? data[siblingKey] : {};
      onDataChange({ ...data, [siblingKey]: { ...prev, [perKey]: newVal } });
      return;
    }
    const arr = Array.isArray(currentValue) ? currentValue.slice() : [];
    const idx = arr.findIndex((sel) => sel.value === optionValue);
    if (idx === -1) return; // shouldn't happen — option not selected
    arr[idx] = { ...arr[idx], [perKey]: newVal };
    onDataChange({ ...data, [field.key]: arr });
  };

  return (
    <div className="structured-multi-choice-block" id={`field-${templateId}-${field.key}`}>
      <label className="structured-field-label">
        {field.label}
        {(field.required || (field.required_when && data?.[field.required_when.field] === field.required_when.equals)) && (
          <span className="required-marker">*</span>
        )}
      </label>
      {field.helperText && <p className="structured-helper-text">{field.helperText}</p>}
      <div className="structured-multi-choice-options">
        {options.map((opt) => {
          const isChecked = isSingle
            ? currentValue === opt.value
            : activeSelections.includes(opt.value);
          const inputType = isSingle ? 'radio' : 'checkbox';
          return (
            <label key={opt.value} className="structured-multi-choice-option">
              <input
                type={inputType}
                name={isSingle ? `field-${templateId}-${field.key}` : undefined}
                checked={isChecked}
                onChange={() => toggleSelection(opt.value)}
              />
              <span>{opt.label ?? opt.value}</span>
            </label>
          );
        })}
      </div>

      {activeSelections.length > 0 && perFields.length > 0 && (
        <div className="structured-multi-choice-per-selection">
          {activeSelections.map((sv) => {
            const opt = options.find((o) => o.value === sv);
            const optLabel = opt?.label ?? sv;
            // Resolve current per-field values for THIS selection
            const subData = isSingle
              ? (data?.[`${field.key}__${sv}`] ?? {})
              : (Array.isArray(currentValue) ? currentValue.find((s) => s.value === sv) ?? {} : {});
            return (
              <div key={sv} className="structured-multi-choice-selection-card">
                <p className="structured-multi-choice-selection-eyebrow">{optLabel}</p>
                {perFields.map((pf) => (
                  <StructuredFieldInput
                    key={pf.key}
                    templateId={templateId}
                    fieldKey={pf.key}
                    parentFieldKey={isSingle ? `${field.key}__${sv}` : `${field.key}-${sv}`}
                    field={pf}
                    value={subData?.[pf.key] ?? ''}
                    onChange={(newVal) => updatePerField(sv, pf.key, newVal)}
                  />
                ))}
              </div>
            );
          })}
        </div>
      )}
    </div>
  );
}
```
