### Mount effect (per RESEARCH.md Pattern 1 + 2)

```javascript
useEffect(() => {
  // Guard 1: invalid partner slug
  if (!VALID_PARTNERS.includes(partner)) {
    navigate('/', { replace: true });
    return;
  }
  // Guard 2: test partner cannot write accountability data (DB CHECK constraint — Pitfall 3)
  if (partner === 'test') {
    navigate(`/hub/${partner}`, { replace: true });
    return;
  }

  Promise.all([
    fetchKpiTemplates(),
    fetchKpiSelections(partner),
    fetchGrowthPriorities(partner),
    fetchGrowthPriorityTemplates(),
  ])
    .then(([tpls, sels, prios, priorityTpls]) => {
      // Guard 3: already locked -> redirect to read-only view (Pattern 2, KPI-06)
      if (sels.length > 0 && sels[0].locked_until) {
        navigate(`/kpi-view/${partner}`, { replace: true });
        return;
      }
      setTemplates(tpls);
      setPriorityTemplates(priorityTpls);
      setExistingSelections(sels);
      setExistingPriorities(prios);
      // Pre-populate in-progress state
      setSelectedTemplateIds(sels.map((s) => s.template_id).filter(Boolean));
      // Map existing growth priorities back to per-slot state
      const personalRow = prios.find((p) => p.type === 'personal');
      const businessRows = prios.filter((p) => p.type === 'business');
      if (personalRow) {
        const matchTpl = priorityTpls.find((t) => t.description === personalRow.description);
        setPersonal(matchTpl
          ? { kind: 'template', templateId: matchTpl.id, customText: '' }
          : { kind: 'custom', templateId: null, customText: personalRow.description });
      }
      if (businessRows[0]) {
        const matchTpl = priorityTpls.find((t) => t.description === businessRows[0].description);
        setBusiness1(matchTpl
          ? { kind: 'template', templateId: matchTpl.id, customText: '' }
          : { kind: 'custom', templateId: null, customText: businessRows[0].description });
      }
      if (businessRows[1]) {
        const matchTpl = priorityTpls.find((t) => t.description === businessRows[1].description);
        setBusiness2(matchTpl
          ? { kind: 'template', templateId: matchTpl.id, customText: '' }
          : { kind: 'custom', templateId: null, customText: businessRows[1].description });
      }
    })
    .catch((err) => {
      console.error(err);
      setLoadError(true);
    })
    .finally(() => setLoading(false));
```

### Handlers

**toggleKpi(templateId)** — per D-03 soft cap:
```javascript
function toggleKpi(templateId) {
  setSelectedTemplateIds((prev) => {
    if (prev.includes(templateId)) return prev.filter((id) => id !== templateId);
    if (prev.length >= 5) return prev; // soft cap: do nothing
    return [...prev, templateId];
  });
}
```

**selectPriorityTemplate(slot, templateId)** / **enableCustom(slot)** / **updateCustomText(slot, text)** — three helpers that take a slot key ('personal' | 'business1' | 'business2') and route to the right setter. Selecting a template clears custom; enabling custom clears template; per D-10 these are mutually exclusive.

**continueToConfirmation()** — validates and PERSISTS selections (per RESEARCH.md Pattern 1 and Pitfall 2 replace-all):
```javascript
async function continueToConfirmation() {
  // Validation
  if (selectedTemplateIds.length !== 5) return;
  const personalValid = (personal.kind === 'template' && personal.templateId)
    || (personal.kind === 'custom' && personal.customText.trim().length > 0);
  const biz1Valid = (business1.kind === 'template' && business1.templateId)
    || (business1.kind === 'custom' && business1.customText.trim().length > 0);
  const biz2Valid = (business2.kind === 'template' && business2.templateId)
    || (business2.kind === 'custom' && business2.customText.trim().length > 0);
  if (!personalValid || !biz1Valid || !biz2Valid) return;

  setSaving(true);
  setSubmitError(null);
  try {
    // Replace-all for kpi_selections: delete any existing non-locked rows first
    for (const row of existingSelections) {
      if (!row.locked_until) {
        // eslint-disable-next-line no-await-in-loop
        await deleteKpiSelection(row.id);
      }
    }
    // Write the 5 new kpi_selections (KPI-05: copy label + category snapshot)
    for (const tid of selectedTemplateIds) {
      const tpl = templates.find((t) => t.id === tid);
      if (!tpl) continue;
      // eslint-disable-next-line no-await-in-loop
      await upsertKpiSelection({
        partner,
        template_id: tpl.id,
        label_snapshot: tpl.label,
        category_snapshot: tpl.category,
        locked_until: null,
      });
    }
    // Replace-all for growth_priorities: delete existing non-locked rows
    // (growth_priorities has no template FK; we store the description text directly)
    for (const row of existingPriorities) {
      if (!row.locked_until) {
        // Delete via direct supabase call is not exposed; use upsert semantics instead.
        // Workaround: use supabase client directly. Import { supabase } alongside the functions.
        // eslint-disable-next-line no-await-in-loop
        await supabase.from('growth_priorities').delete().eq('id', row.id);
      }
    }
    // Insert the 3 new growth_priorities
    const pDesc = personal.kind === 'template'
      ? priorityTemplates.find((t) => t.id === personal.templateId)?.description
      : personal.customText.trim();
    const b1Desc = business1.kind === 'template'
      ? priorityTemplates.find((t) => t.id === business1.templateId)?.description
      : business1.customText.trim();
    const b2Desc = business2.kind === 'template'
      ? priorityTemplates.find((t) => t.id === business2.templateId)?.description
      : business2.customText.trim();

    await upsertGrowthPriority({ partner, type: 'personal', description: pDesc, status: 'active', locked_until: null });
    await upsertGrowthPriority({ partner, type: 'business', description: b1Desc, status: 'active', locked_until: null });
    await upsertGrowthPriority({ partner, type: 'business', description: b2Desc, status: 'active', locked_until: null });

    // Refresh existing rows for accurate re-entry from Back button
    const [freshSels, freshPrios] = await Promise.all([
      fetchKpiSelections(partner),
      fetchGrowthPriorities(partner),
    ]);
    setExistingSelections(freshSels);
    setExistingPriorities(freshPrios);

    setView('confirmation');
  } catch (err) {
    console.error(err);
    setSubmitError(KPI_COPY.selection.errorContinue);
  } finally {
    setSaving(false);
  }
}
```

NOTE: Import `supabase` client alongside the helper functions: `import { supabase, fetchKpiTemplates, ... } from '../lib/supabase.js';`. The inline `supabase.from('growth_priorities').delete()` is needed because `deleteGrowthPriority` is not a separate helper (and adding one is out of scope for this plan — we use the exported client directly, matching the pattern in `supabase.js` where `supabase` is an exported value).

**lockIn()** — per D-07:
```javascript
async function lockIn() {
  setSaving(true);
  setSubmitError(null);
  try {
    const iso = await lockKpiSelections(partner);
    setLockedUntil(iso);
    setView('success');
    setTimeout(() => navigate(`/hub/${partner}`, { replace: true }), 1800);
  } catch (err) {
    console.error(err);
    setSubmitError(KPI_COPY.confirmation.errorLock);
  } finally {
    setSaving(false);
  }
}
```

**backToSelection()** — simply `setView('selection')`. State is preserved in selection state variables.
