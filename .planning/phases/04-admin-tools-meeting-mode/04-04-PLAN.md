```js
// Pattern 6: render-time label fallback for kpi_results (D-06)
function getLabelForEntry(kpiId, entry, lockedKpis) {
  if (entry && entry.label) return entry.label;
  const match = lockedKpis.find((k) => k.id === kpiId);
  return match?.label_snapshot ?? '(unknown KPI)';
}

function motionProps(dir) {
  return {
    initial: { opacity: 0, x: dir * 24 },
    animate: { opacity: 1, x: 0 },
    exit: { opacity: 0, x: dir * -24 },
    transition: { duration: 0.22, ease: 'easeOut' },
  };
}
```

LOAD FLOW (useEffect on mount):

NAVIGATION:
- `goNext()` — if stopIndex < STOPS.length - 1: setDirection(1); setStopIndex(i => i + 1)
- `goPrev()` — if stopIndex > 0: setDirection(-1); setStopIndex(i => i - 1)

NOTE AUTO-SAVE (Pattern 2):
- `handleNoteChange(stopKey, text)`:
  - setNotes((n) => ({ ...n, [stopKey]: text }))
  - if debounceRef.current: clearTimeout(debounceRef.current)
  - debounceRef.current = setTimeout(async () => {
      try {
        await upsertMeetingNote({ meeting_id: id, agenda_stop_key: stopKey, body: text });
        setSavedFlash(stopKey);
        setTimeout(() => setSavedFlash(null), 1500);
      } catch (err) {
        console.error(err);
        setError(MEETING_COPY.errors.noteSaveFail);
      }
    }, DEBOUNCE_MS)

- `handleReflectionChange(partner, kpiId, text)` — debounced similar to notes (400ms). On timer fire, call `adminOverrideScorecardEntry` with current result + new reflection.
